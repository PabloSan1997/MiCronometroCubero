Table Usuario{
  id_user uuid [primary key]
  username varchar(100)
  nickname varchar(100)
  password varchar(100)
  idProb uuid
}

Table Promedio{
  id_prob uuid [primary key]
  avo5 decimal(33002,2)
  media decimal(33002,2)
  mejor decimal(33002,2)
  peor decimal(33002,2)
  idRes uuid
}

Table Resoluciones{
  id_res uuid [primary key]
  algoritmo varchar(60)
  tiempo interger
  fecha Date
  tipo varchar(5)
}

Ref: Promedio.idRes < Resoluciones.id_res
Ref: Usuario.idProb < Promedio.id_prob
